% POCUS Textbook - LaTeX Header for PDF Output
% Chinese language support with xeCJK

% xeCJK package for Chinese support
\usepackage{xeCJK}

% Set Chinese fonts with fallback
\IfFontExistsTF{Microsoft JhengHei}{
  \setCJKmainfont{Microsoft JhengHei}[
    BoldFont = Microsoft JhengHei Bold,
    ItalicFont = Microsoft JhengHei,
    BoldItalicFont = Microsoft JhengHei Bold
  ]
  \setCJKsansfont{Microsoft JhengHei}[
    BoldFont = Microsoft JhengHei Bold
  ]
  \setCJKmonofont{Microsoft JhengHei}
}{
  \IfFontExistsTF{Noto Sans CJK TC}{
    \setCJKmainfont{Noto Sans CJK TC}
    \setCJKsansfont{Noto Sans CJK TC}
    \setCJKmonofont{Noto Sans Mono CJK TC}
  }{
    % Fallback for Linux/GitHub Actions
    \setCJKmainfont{Noto Serif CJK TC}
    \setCJKsansfont{Noto Sans CJK TC}
  }
}

% xeCJK settings
\xeCJKsetup{
  CJKmath=true,
  CheckSingle=true,
  PlainEquation=true
}
